import numpy as np
import pandas as pd
from pandas import Series, DataFrame

1 Merge

# make a dataframe
dframe1 = DataFrame({'key':['x','z','y','z','x','x'], 'value_df1':np.arange(6)})
dframe1

key value_df1
0 x 0
1 z 1
2 y 2
3 z 3
4 x 4
5 x 5 
# make another dataframe
dframe2 = DataFrame({'key':['q','y','z'], 'value_df2':[1, 2, 3]})
dframe2

key value_df2
0 q 1
1 y 2
2 z 3

1.1 Inner Merge

pd.merge(dframe1, dframe2)

key value_df1 value_df2
0 z 1 3
1 z 3 3
2 y 2 2

By default, merge use inner join

pd.merge(dframe1, dframe2, on='key', how='inner')

key value_df1 value_df2
0 z 1 3
1 z 3 3
2 y 2 2

1.2 Lefe Merge

pd.merge(dframe1, dframe2, on='key', how='left')

key value_df1 value_df2
0 x 0 NaN
1 z 1 3.0
2 y 2 2.0
3 z 3 3.0
4 x 4 NaN
5 x 5 NaN

1.3 Right Merge

pd.merge(dframe1, dframe2, on='key', how='right')

key value_df1 value_df2
0 z 1.0 3
1 z 3.0 3
2 y 2.0 2
3 q NaN 1

1.4 Outer Merge

pd.merge(dframe1, dframe2, on='key', how='outer')

key value_df1 value_df2
0 x 0.0 NaN
1 x 4.0 NaN
2 x 5.0 NaN
3 z 1.0 3.0
4 z 3.0 3.0
5 y 2.0 2.0
6 q NaN 1.0

1.5 Merge based on index

df_left = DataFrame({'key': ['X','Y','Z','X','Y'], 'data': range(5)})
df_right = DataFrame({'group_data': [10, 20]}, index=['X', 'Y'])

df_left

key data

1 Y 1
2 Z 2
3 X 3
4 Y 4 
df_right

group_data
X 10
Y 20

1.6 Merge using key from df_left and index from df_right

pd.merge(df_left, df_right, left_on='key', right_index=True)

key data group_data
0 X 0 10
3 X 3 10
1 Y 1 20
4 Y 4 20

1.7 Merge using outer

pd.merge(df_left, df_right, left_on='key', right_index=True, how='outer')

key data group_data
0 X 0 10.0
3 X 3 10.0
1 Y 1 20.0
4 Y 4 20.0
2 Z 2 NaN
